Padding Helpers

.at-no-padding
.at-no-p

Set padding="0" style to an element.

]]>

.at-padding-[5x]
.at-p-[5x]

Set padding="[5x]px" style to an element. Were x goes from 0 to 20

...
]]>

.at-padding-[8x]
.at-p-[8x]

Set padding="[8x]px" style to an element. Were x goes from 1 to 12

...
]]>

.at-padding-[direction]-[5x]
.at-p[dir]-[5x]

Set padding-[direction]="[5x]px" style to an element. Were x goes from 0 to 20.
direction = top | left | right | bottom | horizontal | vertical
dir = t | l | r | b | h | v

...
]]>

.at-padding-[direction]-[8x]
.at-p[dir]-[8x]

Set padding-[direction]="[8x]px" style to an element. Were x goes from 1 to 12.
direction = top | left | right | bottom | horizontal | vertical
dir = t | l | r | b | h | v

...
]]>
